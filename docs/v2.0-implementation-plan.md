> **NOTE:** This is a historical draft. Some features have been implemented in v1.4. The "Browser Close Backup" feature (Gorev 9) has been removed and is no longer planned.

## Overview

**Mevcut Kod Base:**
- `background.js` (507 satir) - Service worker
- `popup.js` (924 satir) - Popup UI logic
- `popup.html` (190 satir) - Popup structure
- `popup.css` (600 satir) - Styling
- `manifest.json` (39 satir) - Manifest V3
- 18 locale dosyasi (her biri ~200 satir)

**Hedef:** 19 yeni ozellik eklemek, 3 fazda.

**Popup Tasarim Degisikligi:**
- Mevcut: `Backup | Restore` (2 tab)
- Yeni: `Backup | Restore | Tools` (3 tab) + settings collapse/expand

### Gorev 8: Akıllı Yedekleme Tetikleyicileri + Hatırlatma

**Dosya:** `background.js` - Tab count threshold:
```javascript
chrome.tabs.onCreated.addListener(checkTabThreshold);

async function checkTabThreshold() {
    const settings = await chrome.storage.local.get([
        CONSTANTS.STORAGE.TAB_THRESHOLD_ENABLED,
        CONSTANTS.STORAGE.TAB_THRESHOLD,
        CONSTANTS.STORAGE.BACKUP_FOLDER_ID
    ]);
    
    if (!settings[CONSTANTS.STORAGE.TAB_THRESHOLD_ENABLED]) return;
    
    const threshold = settings[CONSTANTS.STORAGE.TAB_THRESHOLD] || 20;
    const tabs = await chrome.tabs.query({});
    const validTabs = tabs.filter(t => isValidUrl(t.url));
    
    if (validTabs.length >= threshold) {
        const lastAutoBackup = await chrome.storage.local.get('lastThresholdBackup');
        const oneHourAgo = Date.now() - 3600000;
        if (lastAutoBackup.lastThresholdBackup > oneHourAgo) return;
        
        const folderId = settings[CONSTANTS.STORAGE.BACKUP_FOLDER_ID];
        if (folderId) {
            await performBackup(folderId);
            await chrome.storage.local.set({ lastThresholdBackup: Date.now() });
            showBadge('\u2713', '#4CAF50');
        }
    }
}
```

Periodic reminder:
```javascript
chrome.alarms.create('backupReminder', { periodInMinutes: 60 });

chrome.alarms.onAlarm.addListener((alarm) => {
    if (alarm.name === 'backupReminder') {
        checkBackupReminder();
    }
});

async function checkBackupReminder() {
    const settings = await chrome.storage.local.get([
        CONSTANTS.STORAGE.REMINDER_ENABLED,
        CONSTANTS.STORAGE.REMINDER_DAYS,
        CONSTANTS.STORAGE.LAST_BACKUP_TIME
    ]);
    
    if (!settings[CONSTANTS.STORAGE.REMINDER_ENABLED]) return;
    
    const days = settings[CONSTANTS.STORAGE.REMINDER_DAYS] || 3;
    const lastBackup = settings[CONSTANTS.STORAGE.LAST_BACKUP_TIME] || 0;
    const elapsed = Date.now() - lastBackup;
    const threshold = days * 24 * 60 * 60 * 1000;
    
    if (elapsed > threshold) {
        chrome.action.setBadgeText({ text: '!' });
        chrome.action.setBadgeBackgroundColor({ color: '#FF9800' });
    }
}
```

        // Birden fazla tab kapaniyor, tarayici kapaniyor olabilir

async function attemptCloseBackup() {
    const settings = await chrome.storage.local.get([
        CONSTANTS.STORAGE.BACKUP_FOLDER_ID,
        'closeBackupEnabled'
    ]);
    
    if (!settings.closeBackupEnabled || !settings[CONSTANTS.STORAGE.BACKUP_FOLDER_ID]) return;
    
    try {
        await performBackup(settings[CONSTANTS.STORAGE.BACKUP_FOLDER_ID]);
        await chrome.storage.local.set({ lastCloseBackup: Date.now() });
    } catch (e) {
        console.error('Close backup failed:', e);
    }
    
    tabCloseCount = 0;
}
```

async function restoreTabsList(tabs) {
    const window = await chrome.windows.create({ focused: true });
    let tabsOpened = 0;
    for (const tab of tabs) {
        try {
            const parsed = parseBookmarkTitle(tab.title);
            await chrome.tabs.create({ windowId: window.id, url: tab.url, pinned: parsed.pinned });

    const newTab = allTabs.find(t => t.url === 'chrome://newtab/');

    return { success: true, tabsOpened };
}

    for (const child of children) {
        if (child.url) {
            tabs.push({ title: child.title, url: child.url, date: bookmarkNode[0].title });
        } else {
            const subChildren = await chrome.bookmarks.getChildren(child.id);
            subChildren.forEach(sc => {
                if (sc.url) tabs.push({ title: sc.title, url: sc.url, group: child.title, date: bookmarkNode[0].title });
            });
        }
    }
    
    if (format === 'json') {
        return JSON.stringify({ exportDate: new Date().toISOString(), tabs }, null, 2);
    } else {
        let csv = 'Title,URL,Group,Date\n';
        tabs.forEach(t => {
            csv += `"${t.title}","${t.url}","${t.group || ''}","${t.date}"\n`;
        });
        return csv;
    }
}
```

async function importBackupFromFile(jsonData, folderId) {
    const data = JSON.parse(jsonData);
    if (!data.tabs || !Array.isArray(data.tabs)) throw new Error('Invalid format');
    
    const folderName = generateBackupFolderName('Imported');
    const folder = await chrome.bookmarks.create({ parentId: folderId, title: folderName });
    
    let count = 0;
    for (const tab of data.tabs) {
        if (isValidUrl(tab.url)) {
            await chrome.bookmarks.create({
                parentId: folder.id,
                title: tab.title || tab.url,
                url: tab.url
            });
            count++;
        }
    }
    return count;
}

async function updateBackupStats(parentId) {
    const children = await chrome.bookmarks.getChildren(parentId);
    const backups = children.filter(c => !c.url && c.title?.startsWith('Backup_'));
    
    let totalTabs = 0;
    const domainCounts = {};
    
    for (const backup of backups) {
        const urls = await getAllBookmarkUrls(backup.id);
        totalTabs += urls.length;
        urls.forEach(url => {
            try {
                const domain = new URL(url).hostname;
                domainCounts[domain] = (domainCounts[domain] || 0) + 1;
            } catch {}
        });
    }
    
    const stats = {
        totalBackups: backups.length,
        totalTabs,
        topDomains: Object.entries(domainCounts)
            .sort((a, b) => b[1] - a[1])
            .slice(0, 10),
        lastUpdated: Date.now()
    };
    
    await chrome.storage.local.set({ [CONSTANTS.STORAGE.BACKUP_STATS]: stats });
    return stats;
}
